import { Database } from 'better-sqlite3';

export type Update01Data = {
	value?: number;
}

export type Update01Params = {
	param1: number;
}

export type Update01Result = {
	changes: number;
}

export function update01(db: Database, params: Update01Params): Update01Result {
	const sql = `
	UPDATE mytable1 SET value=? WHERE id=?
	`
	return db.prepare(sql)
		.run([params.value, params.param1]) as Update01Result;
}
